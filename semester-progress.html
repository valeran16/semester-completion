<script>

function updateProgress() {

  const startDate = new Date('2024-01-09');

  const endDate = new Date('2024-05-17');

  const now = new Date();


  const totalDuration = endDate - startDate;

  const timePassed = now - startDate;

  const progressPercentage = Math.min(Math.max(timePassed / totalDuration * 100, 0), 100).toFixed(2);


  document.getElementById('progressBar').style.width = progressPercentage + '%';

  document.getElementById('percentText').innerText = progressPercentage + '% Completed';

}


// Update progress on load

updateProgress();


// Optionally, update the progress every day (not necessary if static)
